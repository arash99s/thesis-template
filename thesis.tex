% تنظیم هدر برای صفحات زوج و فرد

% تنظیم فوتر: شماره صفحه در مرکز پایین
\fancyfoot[C]{\thepage}

% بازتعریف استایل plain برای صفحات شروع فصل‌ها

  \renewcommand{\headrulewidth}{0pt}%

  \fancyhead[LO]{\headerfont \leftmark}%
  % فوتر برای صفحات شروع فصل‌ها
  \fancyfoot[C]{\thepage}%
}
